{{- $resource := .resource -}} {{- $alt := .alt | default "" -}} {{- $class := .class | default "" -}} {{- $loading := .loading | default "lazy" -}} {{- $sizes := .sizes | default "100vw" -}} {{- $maxWidth := .maxWidth | default 1600 -}} {{- $widths := .widths | default (slice 480 768 1080 1440) -}} {{- $decoding := .decoding | default "async" -}} {{- $fetchpriority := .fetchpriority | default (cond (eq $loading "eager") "high" "auto") -}} {{- if not $resource -}} {{- return -}} {{- end -}} {{- if ne $resource.ResourceType "image" -}} {{- return -}} {{- end -}} {{- $processableFormats := slice "jpg" "jpeg" "png" "tif" "bmp" "gif" "webp" -}} {{- if not (in $processableFormats $resource.MediaType.SubType) -}} {{- return -}} {{- end -}} {{- $targetWidth := $resource.Width -}} {{- if gt $targetWidth $maxWidth -}} {{- $targetWidth = $maxWidth -}} {{- end -}} {{- $variantWidths := slice -}} {{- range $widths -}} {{- if le . $targetWidth -}} {{- $variantWidths = $variantWidths | append . -}} {{- end -}} {{- end -}} {{- if not (in $variantWidths $targetWidth) -}} {{- $variantWidths = $variantWidths | append $targetWidth -}} {{- end -}} {{- $fallbackSrcset := slice -}} {{- $webpSrcset := slice -}} {{- $fallbackPrimary := "" -}} {{- range $variantWidths -}} {{- $width := . -}} {{- $fallbackVariant := $resource.Resize (printf "%dx q80" $width) -}} {{- $webpVariant := $resource.Resize (printf "%dx webp q75" $width) -}} {{- if eq $width $targetWidth -}} {{- $fallbackPrimary = $fallbackVariant -}} {{- end -}} {{- $fallbackSrcset = $fallbackSrcset | append (printf "%s %dw" $fallbackVariant.RelPermalink $width) -}} {{- $webpSrcset = $webpSrcset | append (printf "%s %dw" $webpVariant.RelPermalink $width) -}} {{- end -}}
